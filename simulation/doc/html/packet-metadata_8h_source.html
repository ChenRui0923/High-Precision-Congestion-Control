225  struct Data {
226  /* number of references to this struct Data instance. */
227  uint32_t m_count;
228  /* size (in bytes) of m_data buffer below */
229  uint16_t m_size;
230  /* max of the m_used field over all objects which
231  * reference this struct Data instance */
232  uint16_t m_dirtyEnd;
233  /* variable-sized buffer of bytes */
234  uint8_t m_data[PACKET_METADATA_DATA_M_DATA_SIZE];
235  };
236  /* Note that since the next and prev fields are 16 bit integers
237  and since the value 0xffff is reserved to identify the
238  fact that the end or the start of the list is reached,
239  only a limited number of elements can be stored in
240  a m_data byte buffer.
241  */
242  struct SmallItem {
243  /* offset (in bytes) from start of m_data buffer
244  to next element in linked list. value is 0xffff
245  if next element does not exist.
246  stored as a fixed-size 16 bit integer.
247  */
248  uint16_t next;
249  /* offset (in bytes) from start of m_data buffer
250  to previous element in linked list. value is 0xffff
251  if previous element does not exist.
252  stored as a fixed-size 16 bit integer.
253  */
254  uint16_t prev;
255  /* the high 31 bits of this field identify the
256  type of the header or trailer represented by
257  this item: the value zero represents payload.
258  If the low bit of this uid is one, an ExtraItem
259  structure follows this SmallItem structure.
260  stored as a variable-size 32 bit integer.
261  */
262  uint32_t typeUid;
263  /* the size (in bytes) of the header or trailer represented
264  by this element.
265  stored as a variable-size 32 bit integer.
266  */
267  uint32_t size;
268  /* this field tries to uniquely identify each header or
269  trailer _instance_ while the typeUid field uniquely
270  identifies each header or trailer _type_. This field
271  is used to test whether two items are equal in the sense
272  that they represent the same header or trailer instance.
273  That equality test is based on the typeUid and chunkUid
274  fields so, the likelyhood that two header instances
275  share the same chunkUid _and_ typeUid is very small
276  unless they are really representations of the same header
277  instance.
278  stored as a fixed-size 16 bit integer.
279  */
280  uint16_t chunkUid;
281  };
282  struct ExtraItem {
283  /* offset (in bytes) from start of original header to
284  the start of the fragment still present.
285  stored as a variable-size 32 bit integer.
286  */
287  uint32_t fragmentStart;
288  /* offset (in bytes) from start of original header to
289  the end of the fragment still present.
290  stored as a variable-size 32 bit integer.
291  */
292  uint32_t fragmentEnd;
293  /* the packetUid of the packet in which this header or trailer
294  was first added. It could be different from the m_packetUid
295  field if the user has aggregated multiple packets into one.
296  stored as a fixed-size 64 bit integer.
297  */
298  uint64_t packetUid;
299  };

372 PacketMetadata::PacketMetadata (uint64_t uid, uint32_t size)
373  : m_data (PacketMetadata::Create (10)),
374  m_head (0xffff),
375  m_tail (0xffff),
376  m_used (0),
377  m_packetUid (uid)
378 {
379  memset (m_data->m_data, 0xff, 4);

381  {
382  DoAddHeader (0, size);

384 }

394  m_data->m_count++;
395 }
396 PacketMetadata &
397 PacketMetadata::operator = (PacketMetadata const& o)
398 {
399  if (m_data != o.m_data)
400  {
401  // not self assignment
402  NS_ASSERT (m_data != 0);
403  m_data->m_count--;
404  if (m_data->m_count == 0)
405  {
406  PacketMetadata::Recycle (m_data);
407  }
408  m_data = o.m_data;
409  NS_ASSERT (m_data != 0);
410  m_data->m_count++;
411  }
412  m_head = o.m_head;
413  m_tail = o.m_tail;
414  m_used = o.m_used;
415  m_packetUid = o.m_packetUid;
416  return *this;
417 }
418 PacketMetadata::~PacketMetadata ()
419 {
420  NS_ASSERT (m_data != 0);
421  m_data->m_count--;
422  if (m_data->m_count == 0)
423  {
424  PacketMetadata::Recycle (m_data);
425  }
426 }
