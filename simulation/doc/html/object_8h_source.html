362 template <typename T>
363 Ptr<T>
364 Object::GetObject () const
365 {
366  // This is an optimization: if the cast works (which is likely),
367  // things will be pretty fast.
368  T *result = dynamic_cast<T *> (m_aggregates->buffer[0]);
369  if (result != 0)
370  {
371  return Ptr<T> (result);
372  }
373  // if the cast does not work, we try to do a full type check.
374  Ptr<Object> found = DoGetObject (T::GetTypeId ());
375  if (found != 0)
376  {
377  return Ptr<T> (static_cast<T *> (PeekPointer (found)));
378  }
379  return 0;
380 }
