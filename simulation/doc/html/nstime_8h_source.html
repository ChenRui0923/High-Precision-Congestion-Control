432  static inline Time From (const int64x64_t &from, enum Unit timeUnit)
433  {
434  struct Information *info = PeekInformation (timeUnit);
435  // DO NOT REMOVE this temporary variable. It's here
436  // to work around a compiler bug in gcc 3.4
437  int64x64_t retval = from;
438  if (info->fromMul)
439  {
440  retval *= info->timeFrom;
441  }
442  else
443  {
444  retval.MulByInvert (info->timeFrom);
445  }
446  return Time (retval);
447  }
